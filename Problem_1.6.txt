package javaproblem;
import java.util.*;
public class JavaProblem {

    public static void main(String[] args) {

            Scanner input = new Scanner (System.in);
            System.out.print("Enter number : ");
            int n=input.nextInt();
            System.out.print("1 to "+n+" : ");
            for(int i=1;i<=n;i++)
                System.out.print(i+" ");
    }
}
