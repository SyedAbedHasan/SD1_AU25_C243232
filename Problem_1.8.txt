package javaproblem;
import java.util.*;
public class JavaProblem {

    public static void main(String[] args) {
        
            Scanner input = new Scanner (System.in);
            int fact=1;
            System.out.print("Enter number : ");
            int n=input.nextInt();
            System.out.print(n+"! = ");
            for(int i=n;i>=1;i--)
            {   
                fact=fact*i;
            }
            System.out.println(fact);
    }
}
