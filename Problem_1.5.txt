package javaproblem;
import java.util.*;
public class JavaProblem {

    public static void main(String[] args) {

              Scanner input = new Scanner (System.in);
              System.out.print("Enter two numbers : ");
              double num1=input.nextDouble();
              double num2=input.nextDouble();
              System.out.print("Enter operator : ");
              char operation = input.next().charAt(0);
              double result;
              switch(operation){
                  case '+':
                      result=num1+num2;
                      System.out.println("Add two numbers : "+result);
                      break;
                  case '-':
                      result=num1-num2;
                      System.out.println("Subtract two numbers : "+result);
                      break;
                  case '*':
                      result=num1*num2;
                      System.out.println("Multiply two numbers : "+result);
                      break;
                  case '/':
                      result=num1/num2;
                      System.out.println("Divide two numbers : "+result);
                      break;
                  default:
                       System.out.println("Invalid Operator. ");
              }
    }
}
